1   package pongbreaker.gui;
2   
3   import java.awt.Color;
4   import java.awt.Graphics;
5   import java.awt.image.BufferedImage;
6   import javax.swing.JPanel;
7   import pongbreaker.domain.Peliolio;
8   import pongbreaker.peli.Pongbreaker;
9   
10  public class Piirtoalusta extends JPanel implements Paivitettava {
11  
12      private Pongbreaker peli;
13      private BufferedImage img;
14      private Graphics gfx;
15  
16      public Piirtoalusta(Pongbreaker peli) {
17          super.setBackground(Color.black);
18          this.peli = peli;
19          this.img = new BufferedImage(500, 350, BufferedImage.TYPE_BYTE_INDEXED);
20          this.gfx = this.img.createGraphics();
21      }
22  
23      @Override
24      protected void paintComponent(Graphics g) {
25          super.paintComponent(gfx);
26          for (Peliolio piirrettava : this.peli.getPiirrettavat()) {
27              piirrettava.piirra(gfx);
28          }
29          if (!peli.onkoPaalla()) {
30              gfx.drawString("Press spacebar", 200, 175);
31          }
32          g.drawImage(img, 0, 0, this);
33      }
34  
35      @Override
36      public void paivita() {
37          super.repaint();
38      }
39  
40  }
